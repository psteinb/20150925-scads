***
I. Start Sparkcluster
***
#The requested ressources are documented in slum_start_spark.sh. Pleas modify
#this please to meet your needs.
./slurm_start_spark.sh

***
IIa. Interactive Shell
***
start_interactive_shell.sh

***
IIb. Submit Spark Application
***
#URI for master server is stored in $HOME/spark_master.tmp
#your jar should either include all your dependencies or you must ensure
to modify java's class paths to include all your libraries.
./spark-1.3.1/bin/spark-submit --master `cat $HOME/spark_master.tmp` main.jar
